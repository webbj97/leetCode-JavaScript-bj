<h1 align=center>FizzBuzz</h1>

## 解法一（循环）
```js
var fizzBuzz = function(n) {
    var arr = [];
    for (var i = 1; i <= n; i++) {
        if (i % 15 === 0) {
            arr.push("FizzBuzz")
        } else if (i % 5 === 0) {
            arr.push("Buzz")
        } else if (i % 3 == 0) {
            arr.push("Fizz")
        } else {
            arr.push(i.toString())
        }
    }
    return arr
};
```
