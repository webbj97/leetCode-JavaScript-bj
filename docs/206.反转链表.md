## 解法一（倒置）
```js
var reverseList = function(head) {
    let cur = head //  当前节点
    let prev = null //  前置节点初始化（head节点无前置节点）
    while (cur) {
        let temp = cur.next 
        cur.next = prev //调换循序
        prev = cur
        cur = temp
    };
    return prev
};
```
